export function useToggle(init = false) {
  const [v, set] = useState(init);
  return [v, () => set((x) => !x)];
}

// components/Quote.tsx
import { ReactNode } from "react";

export default function Quote({
  children,
  caption,
  icon = "❝",
}: {
  children: ReactNode;
  caption?: string;
  icon?: string;
}) {
  return (
    <figure className="my-6 rounded-xl border bg-neutral-50 p-4 dark:bg-neutral-900">
      <div className="flex gap-3">
        <span className="select-none text-xl leading-none">{icon}</span>
        <blockquote className="prose max-w-none italic dark:prose-invert">
          {children}
        </blockquote>
      </div>
      {caption && (
        <figcaption className="mt-2 pl-7 text-sm opacity-70">
          — {caption}
        </figcaption>
      )}
    </figure>
  );
}
